public class Solution {
    public void sortColors(int[] nums) {
        int len=nums.length;
        int tmp=0;
        //int count1=0,count2=0,count0=0;
        for(int i=0;i<len;i++){
            for(int j=0;j<len-1-i;j++){
                if(nums[j]>nums[j+1]){
                    tmp=nums[j];
                    nums[j]=nums[j+1];
                    nums[j+1]=tmp;
                }
            }
        }
        
        
        
    }
}
