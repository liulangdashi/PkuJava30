public class Solution {
    public void sortColors(int[] nums) {
        int i,j,k,l;
        i=j=k=-1;
        int len=nums.length;
        for(l=0;l<len;l++){
            if(nums[l]==0){
                nums[++k]=2;
                nums[++j]=1;
                nums[++i]=0;
            }
            else if(nums[l]==1){
                nums[++k]=2;
                nums[++j]=1;
            }
            else
                nums[++k]=2;
        }
    }
}
