public class Solution {
     public boolean isHappy(int n) {
        HashSet<Integer> num = new HashSet<Integer>();
        while(n!=1){
            int sum = 0;
            while(n>0){
                sum += (n % 10) * (n % 10);
                n = n / 10;
            }
            if(num.contains(sum)){
                return false;
            } else {
                num.add(sum);
            }
            n = sum;
        }
        return true;
    }
}
