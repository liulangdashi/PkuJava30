public class Solution {
    // you need to treat n as an unsigned value
    public int hammingWeight(int n) {
      /*  if(n<0) return -1;
        if(n>Integer.MAX_VALUE) return 1;
        int count=0;
        do{
            if(n%2==1) count++;
            n=n/2;

        int count = 0;
        for (int i = 0; i < 32; ++i) {
             count+= (n & 1);
            n = n >> 1;//右移一位
        }
        return count;
    }
}
